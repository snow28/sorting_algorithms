#include <iostream>
#include <ctime>
#include <cstdlib>

using namespace std;



void bubble_sort(int ar[],int length){
    bool ready=false;
    while(ready==false){
        ready=true;
        for(int x=0;x<length-1;x++){
            if(ar[x]>ar[x+1]){
                ready=false;
                int temp=ar[x];
                ar[x]=ar[x+1];
                ar[x+1]=temp;
            }
        }
    }
}


void display(int ar[],int length){
for(int x=0;x<length;x++){
    cout<<ar[x]<<endl;
}
}

inline void mySleep(clock_t sec)
{
clock_t start_time = clock();
clock_t end_time = sec * 1000 + start_time;
while(clock() != end_time);
}
