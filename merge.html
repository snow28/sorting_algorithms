Merge Sort

In computer science, merge sort (also commonly spelled mergesort) is an efficient, general-purpose, comparison-based sorting algorithm. Most implementations produce a stable sort, which means that the implementation preserves the input order of equal elements in the sorted output. Mergesort is a divide and conquer algorithm that was invented by John von Neumann in 1945.[1] A detailed description and analysis of bottom-up mergesort appeared in a report by Goldstine and Neumann as early as 1948.

Worst-case perfomance: O(n*logn);

Best-case perfomance: O(n*logn);

Avarage perfomance: O(n*logn);

#include <iostream>
#include <ctime>


using namespace std;

void merge(int myAr[], int start, int center, int last)
{
int *tab_pom = new int[(last-start)];
int i=start,j=center+1,k=0;

while (i <=center && j<=last){
if (myAr[j]<myAr[i]){
    tab_pom[k]=myAr[j];
    j++;
}
else{
    tab_pom[k]=myAr[i];
    i++;
}
    k++;
}
    if (i<=center)
        {
        while(i<=center){
            tab_pom[k]=myAr[i];
            i++;
            k++;
        }
    }
else{
    while(j<=last){
        tab_pom[k]=myAr[j];
        j++;
        k++;
    }
    }
for(i=0;i<=last-start;i++)
myAr[start+i]=tab_pom[i];

//cout << endl;
//for(i=start;i<=last;i++)
//cout<<"intermediate step [" << i << "] = " <<myAr[i]<<endl;
}

void merge_sort(int myAr[], int start, int last){
int center;

if(start!=last){
center=(start+last)/2;
merge_sort(myAr,start,center);
merge_sort(myAr,center+1,last);
merge(myAr,start,center,last);
}
}
